proc cmd command {
	upvar #0 pirate pirate
	upvar #0 closed closed
	switch $closed {
		1 {
			return
		}
	}
	expect {
		"THERE IS A LITTLE AXE HERE.*>" {
			send "get axe\n"
			expect "OK"
			exp_continue
		}
		"There is a little axe here.*\n\\*" {
			send "get axe\n"
			expect "OK"
			exp_continue
		}
		"THREATENING LITTLE DWARVES*\n>" {
			send "throw axe\n"
			expect "\n>"
			send "get axe\n"
			expect "\n>"
			send "throw axe\n"
			exp_continue
		}
		"THREATENING LITTLE*\n>" {
			send "throw axe\n"
			exp_continue
		}
		"threatening little*\n\\*" {
			send "throw axe\n"
			exp_continue
		}
		"REINCARNATE YOU*\n>" {
			send "no\n"
			expect "POINT*.\n"
			exit
		}
		"reincarnate you*\n\\*\\*" {
			send "no\n"
			expect "point*.\n"
			exit
		}
		"THE CAVE IS NOW CLOSED" {
			upvar #0 closed closed
			set closed 1
		}
		"The cave is now closed" {
			upvar #0 closed closed
			set closed 1
		}
		"HAR, HAR" {
			switch $pirate {
				0 {
					expect "\n>"
					send "quit\n"
					expect "\n>"
					send "y\n"
					expect "POINT*.\n"
					exit
				}
				default {
					exp_continue
				}
			}
		}
		"\n>" { send "${command}\n" }
		"\n\\*" { send "${command}\n" }
		"\n\\*\\*" { send "${command}\n" }
	}
}
