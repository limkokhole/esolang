					expect "\n>"

				}

					expect "\n>"
					send "throw axe\n"
					expect "\n>"
					send "get axe\n"
					exp_continue
				}
			}
		}
		"REINCARNAT" {
			upvar #0 reincarnate reincarnate
			switch $reincarnate {
				1 {
					set reincarnate 2
				}
				default {
					expect "\n>"
					send "no\n"
					expect "POINT*.\n"
					exit
				}
			}
		}
		"THE CAVE IS NOW CLOSED" {
			upvar #0 closed closed
			set closed 1
		}
		"HAR, HAR" {
			upvar #0 pirate pirate
			switch $pirate {
				1 {
					set pirate 2
					exp_continue
				}
				default {
					expect "\n>"
					send "quit\n"
					expect "\n>"
					send "y\n"
					expect "POINT*.\n"
					exit
				}
			}
		}
		"\n>" { send "${command}\n" }
	}
}
